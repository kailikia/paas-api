STEP 1
sudo apt update
sudo apt install inotify-tools supervisor

STEP 2
sudo chmod +x /var/www/paas-api/app_rebuild.sh
sudo chown -R root:root /var/www/paas/deployed_apps
sudo chown -R root:root /var/www/paas-api/app_rebuild.sh

STEP 3
nano /etc/supervisor/conf.d/monitor_rebuild_changes.conf

[program:monitor_rebuild_changes]
command=/bin/bash -c "/usr/bin/inotifywait -m -e create /var/www/paas/rebuild-report | /bin/bash /var/www/paas-api/app_rebuild.sh"
autostart=true
autorestart=true
stderr_logfile=/var/www/paas/logs/rebuild_supervisor_stderr.log
stdout_logfile=/var/www/paas/logs/rebuild_supervisor_stdout.log

STEP 4
sudo supervisorctl reread
sudo supervisorctl update
